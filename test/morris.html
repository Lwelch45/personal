Updating data



function data(offset) {
  var ret = [];
  for (var x = -9; x <= 11; x += 1) {

    if(x % 6 < 3){
        ret.push({
            x: Math.cos(x).toFixed(4),
            z: 0,
            z: Math.sin(x).toFixed(4)
        });
    }else{
       ret.push({
            x: Math.cos(x).toFixed(4),
            y: Math.sin(x).toFixed(4),
            z: 0
        });
    }
  }
  return ret;
}
var graph = Morris.Line({
    element: 'graph',
    data: data(0),
    xkey: 'x',
    ykeys: ['y', 'z'],
    labels: ['positive', 'public'],
    parseTime: false,
    ymin: -1.0,
    ymax: 1.0,
    hideHover: true
});
